nfl_data <- all_nfl_data |>
  select(PlayType, posteam, yards_gained, down, qtr, 
         ydstogo, ScoreDiff, Season, PassOutcome, DefensiveTeam) |>
    filter(!PlayType %in% c("End of Game", 
                          "Extra Point", 
                          "Half End", 
                          "Kickoff", 
                          "Quarter End",
                          "Two Minute Warning",
                          "Timeout",
                          "No Play"))
CAR2015_play_type_table <- nfl_data |>
  filter(posteam == "CAR", Season == "2015") |>
  count(PlayType) |>
  gt() |>
  tab_header(
    title = "Frequency of Play Types"
  ) |>
  cols_label(
    PlayType = "Play Type",
    n = "Count"
  ) |>
  tab_style(
    style = cell_text(weight = "bold"),
    locations = cells_column_labels(everything())

  opt_row_striping()

yds_sum_table <- CAR2015_play_down_data |>
  group_by(PlayType) |>
  summarize(
    Mean = mean(yards_gained, na.rm = TRUE),
    Median = median(yards_gained, na.rm = TRUE),
    SD = sd(yards_gained, na.rm = TRUE),
    Minimum = min(yards_gained, na.rm = TRUE),
    Maximum = max(yards_gained, na.rm = TRUE),
    Count = n(),
    .groups = "drop"
  ) |>
  gt() |>
  fmt_number(
    columns = c(Mean, Median, SD, Minimum, Maximum),
    decimals = 4
  ) |>
  tab_header(
    title = "Yards Gained Summary by Play Type"
  )
